\section*{Collect}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Dómine exáudi oratiónem meam.\\
	\Rbar.~Et clamor meus ad te véniat.
	
	Orémus.
	Omnípotens sempitérne Deus, qui cæléstia simul et terréna moderáris:
	supplicatiónes pópuli tui cleménter exáudi;
	et pacem tuam nostris concéde tempóribus.
	Per Dóminum.
	\Rbar.~Amen.
}{
	Lord, hear my prayer.
	\Rbar.~And let my cry come unto Thee.
	
	Let us pray.
	Almighty and everlasting God, who dost govern all things in heaven and earth,
	mercifully hear the prayers of Thy people,
	and grant us Thy peace all the days of our life.
	Through our Lord.
	\Rbar.~Amen.
}

\end{latinenglishsection}

\section*{Commemoration of St. Paul the First Hermit (Jan. 14)}

\textit{\textnormal{Ant.} This man, despising the world, and triumphing over earthly things, hath laid up treasures in heaven by word and deed.} %PT: alleluia

\gresetinitiallines{1}
\gregorioscore{../commemorations/conf-non-pont-2v}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Justum dedúxit Dóminus per vias rectas.\\ %PT: allelúia
	\Rbar.~Et osténdit illi regnum Dei.\\  %PT: allelúia
	
	Orémus.
	Deus, qui nos beáti Paul Confessóris tui ánnua solemnitáte lætíficas:
	concéde propítius: ut cujus natalítia cólimus,
	étiam actiónes imitémur.
	Per Dóminum.
}{
	The Lord led the just in right paths. %PT: alleluia
	\Rbar.~And showed him the kingdom of God. %PT: alleluia
	
	Let us pray.
	O God, who dost gladden us by the annual solemnity of blessed Paul, thy Confessor,
	mercifully grant that as we celebrate his heavenly birth, we may also imitate his example.
	Through Our Lord.
}

\end{latinenglishsection}

\section*{Commemoration of Pope St. Marcellus I, Martyr}

\textit{\textnormal{Ant.} This Saint fought even unto death for the law of his God, and feared not the words of the wicked; for he was founded upon a firm rock.}

\gresetinitiallines{1}
\gregorioscore{../commemorations/one-martyr-1v}

\vfill\pagebreak

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Glória et honóre coronásti eum, Dómine.\\
	\Rbar.~Et constituísti eum super ópera mánuum tuárum.\\
	
	Orémus.
	Preces pópuli tui, qu\'{\ae}sumus Dómine, cleménter exáudi:
	ut beáti Marcélli Mártyris tui atque Pontíficis méritis adjuvémur,
	cujus passióne lætámur.
	Per Dóminum.
}{
	Thou hast crowned him with glory and honor, O Lord.
	\Rbar.~And hast set him over the works of Thy hands.
	
	Let us pray.
	We beseech Thee, O Lord, mercifully hear the prayers of Thy people,
	that we may be aided by the merits of blessed Marcellus, Thy martyr and Bishop,
	in whose martyrdom we rejoice.
	Through Our Lord.
}

\end{latinenglishsection}
