\section*{Collect}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Dómine exáudi oratiónem meam.\\
	\Rbar.~Et clamor meus ad te véniat.
	
	Orémus.
	Omnípotens sempitérne Deus, infirmitátem nostram propítius réspice:

\section*{Commemoration of St. Raymond of Penafort, Confessor (Jan. 23)}

\textit{\textnormal{Ant.} This man, despising the world, and triumphing over earthly things, hath laid up treasures in heaven by word and deed.} %PT: alleluia

\gresetinitiallines{1}
\gregorioscore{../commemorations/conf-non-pont-2v}
% PT: \gregorioscore{../commemorations/conf-non-pont-2v-pt}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Justum dedúxit Dóminus per vias rectas.\\ %PT: allelúia
	\Rbar.~Et osténdit illi regnum Dei.\\  %PT: allelúia
	
	Orémus.
	
	Deus, qui beátum Raymúndum pœniténtiæ sacraménti insígnem minístrum elegísti, et per maris undas mirabíliter traduxísti: concéde; ut ejus intercessióne dignos pœniténtiæ fructus fácere, et ad ætérnæ salútis portum perveníre valeámus.
	
	Per Dóminum.
}{
	The Lord led the just in right paths. %PT: alleluia
	\Rbar.~And showed him the kingdom of God. %PT: alleluia
	
	Let us pray.
	
	O God, Who didst choose the blessed Raymond to be an eminent minister of the Sacrament of Penance, and in a wonderful manner didst make him to pass over the waves of the sea, grant unto us, at his petition, the grace to bring forth fruits worthy of repentance, and in the end to attain unto the harbour of eternal salvation.
	
	Through Our Lord.
}

\end{latinenglishsection}

\section*{Commemoration of St. Emerentiana, Virgin Martyr (Jan. 23)}

\textit{\textnormal{Ant.} He that will come after Me, let him deny himself, and take up his cross, and follow Me.}

\gresetinitiallines{1}
\gregorioscore{../commemorations/one-martyr-2v}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Justus ut palma florébit.\\
	\Rbar.~Sicut cedrus Líbani sic multiplicábitur.\\
	
	Orémus.
	
	Indulgéntiam nobis, qu\'{\ae}sumus, Dómine, beáta Emerentiána Virgo et Martyr implóret: quæ tibi grata semper éxstitit, et mérito castitátis, et tuæ professióne virtútis.
	
	Per Dóminum.
}{
	The just shall flourish like the palm tree.
	\Rbar.~And shall grow up like the cedar of Libanus.
	
	Let us pray.
	
	O Lord, we pray thee that thy blessed handmaiden and witness Emerentiana in whom Thou hast ever been well pleased, because of her worthy maidenhood and her mighty showing-forth of thy power, may entreat thy pardon on our behalf.
	
	Through Our Lord.
}
